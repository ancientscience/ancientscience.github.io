int firstkriya(int beta, int alpha)
{
    // max no of Kriyas with beta Pranayams
    int n = std::min(beta, alpha); 

    std::vector<int> f(n, 0);

    f[0] = 1;

    for(int p = 1; p <= beta; p++)
    {
        int prev = 0, cur = 0;

        for(int k = 0; k < n; k++)
        {   
            cur = f[k];
            f[k] += prev + (k+1 < n ? f[k+1] : 0);
            prev = cur;
        }
    }

    return f[0];
}
